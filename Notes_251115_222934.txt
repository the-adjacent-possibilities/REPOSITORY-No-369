# **STEP 2: FIX APK & NETWORK SCRIPTS**

```bash
# Create AetherMind scripts
cat > ~/AetherMind/analyze_apks.sh << 'EOF'
#!/bin/bash
echo "[APK] Scanning /sdcard/Download/*.apk"
mkdir -p ~/AetherMind/logs
for apk in /sdcard/Download/*.apk; do
    [ -f "$apk" ] || continue
    echo "[APK] Analyzing: $(basename "$apk")"
    apktool d "$apk" -o "/tmp/apk_$(basename "$apk")" 2>/dev/null
    echo "[APK] $(basename "$apk") → decompiled" >> ~/AetherMind/logs/apk_analysis.log
done
echo "[APK] Analysis complete."
EOF
chmod +x ~/AetherMind/analyze_apks.sh

# Install apktool
pkg install apktool
```
