import matplotlib.pyplot as plt
from scipy.fft import fft, fftfreq
from scipy.stats import norm
import sympy as sp
from scipy.optimize import minimize
import pandas as pd  # For data export, if your pipeline hungers

# Tier 1: Core Generator - Synthetic Trials with Torsion Signal
def generate_torsion_signal(N_trials=10000, dt=1e-4, T=0.2, L=0.01, f0=10.0, FQC_true=0.97):
    """
    Generate acceleration time series for torsion detection.
    a_torsion = phi^3 * (G/c^2) * |grad_ESQET|^2 * L * FQC^4
    Waveform in Fourier: sum_{n=0}^5 phi^{-n} exp(i 2pi f tau_n), tau_n = n / f0
    Noise: gyro sensitivity 1e-19 rad/s/sqrt(Hz), folded to m/s^2 via L.
    """
    phi = (1 + sp.sqrt(5))/2  # Golden ratio, exact
    G_over_c2 = 7.426e-28  # m/kg
    grad_ESQET = 1e15  # m^{-2}, your baseline
    amp_base = float(phi**3 * G_over_c2 * grad_ESQET**2 * L * FQC_true**4)  # ~3.25e-18 * FQC^4
    
    fs = 1/dt  # Sampling freq, 10 kHz for 100 ms ramp + 200 ms drift
    t = np.linspace(0, T, int(T/dt))
    freqs = fftfreq(len(t), dt)
    
    signals = []
    for _ in range(N_trials):
        # Template waveform in freq domain
        tau_n = np.arange(6) / f0
        template = np.zeros_like(freqs, dtype=complex)
        for n in range(6):
            mask = np.abs(freqs - (n * f0)) < 0.5  # Bin around harmonics
            template += float(phi**(-n)) * np.exp(1j * 2 * np.pi * freqs * tau_n[n]) * mask
        a_signal = np.real(np.fft.ifft(template * amp_base * len(t)))[:len(t)]  # IFFT to time
        
        # Add Gaussian noise: sigma from gyro sens * sqrt(bw * T), bw=fs/2
        sigma_gyro = 1e-19 * np.sqrt(fs/2 * T) * L  # rad -> m/s^2
        noise = np.random.normal(0, sigma_gyro, len(t))
        a_total = a_signal + noise
        signals.append(a_total)
    
    return np.array(signals), t, freqs, amp_base

def bayesian_fqc_inference(signals, t, freqs, prior_mu=0.95, prior_sigma=0.05, lisa_residuals=None):
    """
    Bayesian fit for FQC: posterior on FQC and |grad_ESQET| via likelihood on Fourier peaks.
    Log-posterior: log L(data | theta) + log prior(theta)
    Optimize with scipy; cross-corr with LISA if provided (mock residuals as array).
    Returns: FQC_est, grad_est, snr
    """
    def log_likelihood(theta, signal_fft):
        FQC, grad = theta
        phi = (1 + sp.sqrt(5))/2
        G_over_c2 = 7.426e-28
        L = 0.01
        amp_model = float(phi**3 * G_over_c2 * grad**2 * L * FQC**4)
        
        # Model template FFT
        f0 = 10.0
        tau_n = np.arange(6) / f0
        template_model = np.zeros_like(freqs, dtype=complex)
        for n in range(6):
            mask = np.abs(freqs - (n * f0)) < 0.5
            template_model += float(phi**(-n)) * np.exp(1j * 2 * np.pi * freqs * tau_n[n]) * mask
        model_fft = template_model * amp_model * len(t)
        
        # Chi^2 in freq domain
        chi2 = np.sum(np.abs(np.fft.fft(signal_fft) - model_fft)**2 / (1e-19 * L)**2)  # Noise var
        return -0.5 * chi2
    
    def log_prior(theta):
        FQC, grad = theta
        if 0.5 < FQC < 1.0 and 1e14 < grad < 1e16:
            return norm.logpdf(FQC, prior_mu, prior_sigma) + norm.logpdf(np.log10(grad), 15, 0.5)
        return -np.inf
    
    def neg_log_post(theta, signal_fft):
        return -(log_likelihood(theta, signal_fft) + log_prior(theta))
    
    results = []
    for sig in signals[:100]:  # Subsample for demo; scale to 10^4
        res = minimize(neg_log_post, [0.95, 1e15], args=(sig,), method='L-BFGS-B',
                       bounds=[(0.5,1.0), (1e14,1e16)])
        FQC_est, grad_est = res.x
        # SNR: amp / sigma
        snr = float(( (1 + sp.sqrt(5))/2 )**3 * 7.426e-28 * grad_est**2 * 0.01 * FQC_est**4 ) / (1e-19 * 0.01)
        results.append((FQC_est, grad_est, snr, res.fun))  # -logpost for chain
    
    df = pd.DataFrame(results, columns=['FQC_est', 'grad_ESQET_est', 'SNR', 'neg_logpost'])
    
    # Cross-corr with LISA (mock: synthetic residuals ~1e-15 m/s^2)
    if lisa_residuals is not None:
        corr = np.corrcoef(df['SNR'], lisa_residuals[:len(df)])[0,1]
        df['lisa_corr'] = corr
    
    return df
